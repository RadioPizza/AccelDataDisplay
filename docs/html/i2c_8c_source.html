9uint8_t I2C_Init(I2C_Mode_t mode)
10{
11 uint16_t CCR;
12 uint8_t CCRL_Value;
13 uint8_t CCRH_Value;
14 uint8_t TRISER_Value;
15
16 /* Включаем тактирование периферии I2C */
17 CLK_PCKENR1 |= 0x02; // Бит 1 соответствует I2C в PCKENR1
18
19 /* Настраиваем частоту I2C */
20 I2C_FREQR = 16; // Частота тактирования 16 МГц
21
22 /* Настраиваем параметры в зависимости от режима */
23 if (mode == I2C_STANDARD_MODE) {
24 /* Стандартный режим (100 кГц) */
25 CCRL_Value = 0x50;
26 CCRH_Value = 0x00;
27 TRISER_Value = 17;
28 }
29 else { /* I2C_FAST_MODE */
30 /* Быстрый режим (400 кГц) */
31 CCR = 13; // Вычислено ранее
32 CCRL_Value = (uint8_t)(CCR & 0xFF); // Младшие 8 бит CCR
33 CCRH_Value = (uint8_t)((CCR >> 8) & 0x0F) | I2C_CCRH_FS; // Старшие биты CCR + бит FS (Fast Mode)
34 TRISER_Value = 6;
35 }
36
37 /* Устанавливаем CCRL и CCRH */
38 I2C_CCRL = CCRL_Value;
39 I2C_CCRH = CCRH_Value;
40
41 /* Устанавливаем максимальное время нарастания */
42 I2C_TRISER = TRISER_Value;
43
44 /* Сбрасываем регистр управления 1 */
45 I2C_CR1 = 0x00;
46
47 /* Устанавливаем бит ACK в регистре CR2 */
48 I2C_CR2 |= I2C_CR2_ACK;
49
50 /* Устанавливаем собственный адрес (не используется в режиме мастера) */
51 I2C_OARL = 0x00;
52 I2C_OARH = 0x00; // В 7-битном режиме адресации бит ADDMODE должен быть сброшен
53
54 /* Включаем периферию I2C */
55 I2C_CR1 |= I2C_CR1_PE; // Устанавливаем бит PE для включения I2C
56
57 /* Проверяем успешность инициализации */
58 return (I2C_CR1 & I2C_CR1_PE) ? 0 : 1;
59}
